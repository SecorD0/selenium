def self.new(**opts)
  listener = opts.delete(:listener)
  bridge = Bridge.handshake(opts)
  if bridge.dialect == :w3c
    W3C::Driver.new(bridge, listener: listener)
  else
    OSS::Driver.new(bridge, listener: listener)
  end
end
