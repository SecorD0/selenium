def respond(code: 200, headers: {}, body: '')
  @devtools.fetch.fulfill_request(
    request_id: @id,
    body: Base64.strict_encode64(body),
    response_code: code,
    response_headers: headers.map do |k, v|
      {name: k, value: v}
    end
  )
end
