def install_addon(path, temporary)
  addon = File.open(path, 'rb') { |crx_file| Base64.strict_encode64 crx_file.read }

  payload = {addon: addon}
  payload[:temporary] = temporary unless temporary.nil?
  execute :install_addon, {}, payload
end
