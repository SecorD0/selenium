def initialize(source, duration, delta_x, delta_y, origin: VIEWPORT, x: 0, y: 0)
  super(source)
  @type = :scroll
  @duration = duration * 1000
  @origin = origin
  @x_offset = x
  @y_offset = y
  @delta_x = delta_x
  @delta_y = delta_y
end
