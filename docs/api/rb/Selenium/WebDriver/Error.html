def self.for_code(code)
  case code
  when nil, 0
    nil
  when Integer
    ERRORS.fetch(code)
  when String
    klass_name = code.split(' ').map(&:capitalize).join.sub(/Error$/, '')
    const_get("#{klass_name}Error")
  end
rescue KeyError, NameError
  WebDriverError
end
