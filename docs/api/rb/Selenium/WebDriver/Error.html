def for_code(code)
  return if [nil, 0].include? code
  return ERRORS[code - 1] if code.is_a? Integer

  klass_name = code.split(' ').map(&:capitalize).join
  Error.const_get("#{klass_name.gsub('Error', '')}Error")
rescue NameError
  WebDriverError
end
